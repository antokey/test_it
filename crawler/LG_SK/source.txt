// radio 효과
function customRadio(radioName){
	var radioButton = $('input[name="'+ radioName +'"]');
	$(radioButton).each(function(){
		if($(this).is(':checked')){
			$(this).parent().parent().parent().addClass("selected");
		}
	});
	$(radioButton).click(function(){
		if($(this).is(':checked')){
			$(this).parent().parent().parent().addClass("selected");
		}
		$(radioButton).not(this).each(function(){
			$(this).parent().parent().parent().removeClass("selected");
		});
	});
}

//요금제 선택 시(레이어팝업)
function choiceSvc() {
	var formObj = document.itemForm;
	var svcCodeHtml = '';
	
	if(formObj.checkProdCd != null) {
		if(formObj.checkProdCd.length != undefined) {
			for(var i=0; i &lt; formObj.checkProdCd.length; i++) {
				if(formObj.checkProdCd[i].checked == true) {
					svcCodeHtml = "&lt;a href=\""+formObj.checkProdUrl[i].value+"\" class=\"pointcolor01\"&gt;"+formObj.checkProdNm[i].value+"&lt;/a&gt;";
					formObj.svcCd1.value = formObj.checkProdCd[i].value;
					formObj.svcNm1.value = formObj.checkProdNm[i].value;
				}
			}
		}else{
			if(formObj.checkProdCd.checked == true) {
				svcCodeHtml = "&lt;a href=\""+formObj.checkProdUrl.value+"\" class=\"pointcolor01\"&gt;"+formObj.checkProdNm.value+"&lt;/a&gt;";
				formObj.svcCd1.value = formObj.checkProdCd.value;
				formObj.svcNm1.value = formObj.checkProdNm.value;
			}
		}
	}
	if(svcCodeHtml == '') {
		alert("요금제를 선택하신 후 이용하실수 있습니다.");
		return;
	}
	
	$('#layer_checkprice').html(svcCodeHtml);
	//document.getElementById('lyPop_wrap').style.display = 'none';
	//$('html').removeAttr('style');
	$('#modalPop_wrap').modal('hide');
}
